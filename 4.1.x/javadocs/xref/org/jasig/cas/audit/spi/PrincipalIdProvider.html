19  package org.jasig.cas.audit.spi;
20  
21  import org.jasig.cas.authentication.Authentication;
22  
23  /**
24   * Strategy interface to provide principal id tokens from any given authentication event.
25   *
26   * Useful for authentication scenarios where there is not only one primary principal id available, but additional authentication metadata
27   * in addition to custom requirement to compute and show more complex principal identifier for auditing purposes.
28   * An example would be compound ids resulted from multi-legged mfa authentications, 'surrogate' authentications, etc.
29   *
30   * @author Dmitriy Kopylenko
31   * @since 4.1.9
32   */
33  public interface PrincipalIdProvider {
34  
35      /**
36       * Return principal id from a given authentication event.
37       *
38       * @param authentication authentication event containing the data to computed the final principal id from
39       *
40       * @return computed principal id
41       */
42      String getPrincipalIdFrom(Authentication authentication);
43  }
