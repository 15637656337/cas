43  /**
44   * Tests for {@link org.jasig.cas.adaptors.jdbc.SearchModeSearchDatabaseAuthenticationHandler}.
45   *
46   * @author Misagh Moayyed mmoayyed@unicon.net
47   * @since 4.0.0
48   */
49  @RunWith(SpringJUnit4ClassRunner.class)
50  @ContextConfiguration("classpath:/jpaTestApplicationContext.xml")
51  public class SearchModeSearchDatabaseAuthenticationHandlerTests {
52  
53      private SearchModeSearchDatabaseAuthenticationHandler handler;
54  
55      @Autowired
56      private DataSource dataSource;
57  
58      @Before
59      public void setup() throws Exception {
60          this.handler = new SearchModeSearchDatabaseAuthenticationHandler();
61          handler.setDataSource(this.dataSource);
62          handler.setTableUsers("cassearchusers");
63          handler.setFieldUser("username");
64          handler.setFieldPassword("password");
65          handler.afterPropertiesSet();
66  
67          final Connection c = this.dataSource.getConnection();
68          final Statement s = c.createStatement();
69          c.setAutoCommit(true);
70  
71          s.execute(getSqlInsertStatementToCreateUserAccount(0));
72          for (int i = 0; i < 10; i++) {
73              s.execute(getSqlInsertStatementToCreateUserAccount(i));
74          }
75  
76          c.close();
77      }
78  
79      @After
80      public void tearDown() throws Exception {
81          final Connection c = this.dataSource.getConnection();
82          final Statement s = c.createStatement();
83          c.setAutoCommit(true);
84  
85          for (int i = 0; i < 5; i++) {
86              final String sql = String.format("delete from casusers;");
87              s.execute(sql);
88          }
89          c.close();
90      }
91  
92      private String getSqlInsertStatementToCreateUserAccount(final int i) {
93          return String.format("insert into cassearchusers (username, password) values('%s', '%s');", "user" + i, "psw" + i);
94      }
95  
96      @Entity(name="cassearchusers")
97      public static class UsersTable {
98          @Id
99          @GeneratedValue(strategy = GenerationType.AUTO)
100         private Long id;
101 
102         private String username;
103         private String password;
104     }
105 
106     @Test(expected = FailedLoginException.class)
107     public void verifyNotFoundUser() throws Exception {
108         final UsernamePasswordCredential c = TestUtils.getCredentialsWithDifferentUsernameAndPassword("hello", "world");
109         this.handler.authenticateUsernamePasswordInternal(c);
110     }
111 
112     @Test
113     public void verifyFoundUser() throws Exception {
114         final UsernamePasswordCredential c = TestUtils.getCredentialsWithDifferentUsernameAndPassword("user3", "psw3");
115         assertNotNull(this.handler.authenticateUsernamePasswordInternal(c));
116     }
117 
118     @Test
119     public void verifyMultipleUsersFound() throws Exception {
120         final UsernamePasswordCredential c = TestUtils.getCredentialsWithDifferentUsernameAndPassword("user0", "psw0");
121         assertNotNull(this.handler.authenticateUsernamePasswordInternal(c));
122     }
123 
124 }
